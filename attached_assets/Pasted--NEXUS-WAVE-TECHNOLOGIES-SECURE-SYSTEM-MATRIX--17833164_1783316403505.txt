## 4. FIREBASE GOOGLE CREDENTIAL MANAGER (Website Backend Secrets Setup)
# App ke andar koi code nahi badalna hai, bas website backend config ke andar Client ID point karni hai.
# Android Credential Manager app se token lekar website backend validation API ko pass karega.

```javascript
// WEBSITE BACKEND (Secure Server Endpoint for Google Identity / Credential Manager)
const { OAuth2Client } = require('google-auth-library');

// Client ID strictly backend secrets (.env) ke andar locked rahegi
const GOOGLE_CLIENT_ID = process.env.GOOGLE_WEB_CLIENT_ID; 
const client = new OAuth2Client(GOOGLE_CLIENT_ID);

exports.verifyGoogleUser = async (req, res) => {
  try {
    const { idToken } = req.body; // App ke Credential Manager se aaya hua token
    
    const ticket = await client.verifyIdToken({
        idToken: idToken,
        audience: GOOGLE_CLIENT_ID,
    });
    
    const payload = ticket.getPayload();
    const userId = payload['sub']; // Unique Profile ID for Detection
    
    // Auto login/register profile session
    res.status(200).json({ success: true, user: { uid: userId, email: payload.email, name: payload.name } });
  } catch (error) {
    res.status(401).json({ success: false, error: "Invalid Google Credentials" });
  }
};
